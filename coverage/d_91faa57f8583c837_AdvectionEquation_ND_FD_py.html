7class advectionNd(GenericNDimFinDiff): 

8 r""" 

9 Example implementing the unforced ND advection equation with periodic 

10 or Dirichlet boundary conditions in :math:`[0,1]^N`, 

11 and initial solution of the form 

12 

13 .. math:: 

14 u({\bf x},0) = \prod_{i=1}^N \sin(f\pi x_i), 

15 

16 with :math:`x_i` the coordinate in :math:`i^{th}` dimension. 

17 Discretization uses central finite differences. 

18 

19 Parameters 

20 ---------- 

21 nvars : int of tuple, optional 

22 Spatial resolution (same in all dimensions). Using a tuple allows to 

23 consider several dimensions, e.g nvars=(16,16) for a 2D problem. 

24 c : float, optional 

25 Advection speed (same in all dimensions). 

26 freq : int of tuple, optional 

27 Spatial frequency :math:`f` of the initial conditions, can be tuple. 

28 stencil_type : str, optional 

29 Type of the finite difference stencil. 

30 order : int, optional 

31 Order of the finite difference discretization. 

32 lintol : float, optional 

33 Tolerance for spatial solver (GMRES). 

34 liniter : int, optional 

35 Max. iterations number for GMRES. 

36 solver_type : str, optional 

37 Solve the linear system directly or using GMRES or CG 

38 bc : str, optional 

39 Boundary conditions, either "periodic" or "dirichlet". 

40 sigma : float, optional 

41 If freq=-1 and ndim=1, uses a Gaussian initial solution of the form 

42 

43 .. math:: 

44 u(x,0) = e^{ 

45 \frac{\displaystyle 1}{\displaystyle 2} 

46 \left( 

47 \frac{\displaystyle x-1/2}{\displaystyle \sigma} 

48 \right)^2 

49 } 

50 

51 Attributes 

52 ---------- 

53 A: sparse matrix (CSC) 

54 FD discretization matrix of the ND grad operator. 

55 Id: sparse matrix (CSC) 

56 Identity matrix of the same dimension as A 

57 

58 Note 

59 ---- 

60 Args can be set as values or as tuples, which will increase the dimension. 

61 Do, however, take care that all spatial parameters have the same dimension. 

62 """ 

63 

64 def __init__( 

65 self, 

66 nvars=512, 

67 c=1.0, 

68 freq=2, 

69 stencil_type='center', 

70 order=2, 

71 lintol=1e-12, 

72 liniter=10000, 

73 solver_type='direct', 

74 bc='periodic', 

75 sigma=6e-2, 

76 ): 

77 super().__init__(nvars, -c, 1, freq, stencil_type, order, lintol, liniter, solver_type, bc) 

78 

79 if solver_type == 'CG': # pragma: no cover 

80 self.logger.warn('CG is not usually used for advection equation') 

81 self._makeAttributeAndRegister('c', localVars=locals(), readOnly=True) 

82 self._makeAttributeAndRegister('sigma', localVars=locals()) 

83 

84 def u_exact(self, t, **kwargs): 

85 """ 

86 Routine to compute the exact solution at time t 

87 

88 Parameters 

89 ---------- 

90 t : float 

91 Time of the exact solution. 

92 **kwargs : dict 

93 Additional arguments (that won't be used). 

94 

95 Returns 

96 ------- 

97 sol : dtype_u 

98 The exact solution. 

99 """ 

100 # Initialize pointers and variables 

101 ndim, freq, c, sigma, sol = self.ndim, self.freq, self.c, self.sigma, self.u_init 

102 

103 if ndim == 1: 

104 x = self.grids 

105 if freq[0] >= 0: 

106 sol[:] = np.sin(np.pi * freq[0] * (x - c * t)) 

107 elif freq[0] == -1: 

108 # Gaussian initial solution 

109 sol[:] = np.exp(-0.5 * (((x - (c * t)) % 1.0 - 0.5) / sigma) ** 2) 

110 

111 elif ndim == 2: 

112 x, y = self.grids 

113 sol[:] = np.sin(np.pi * freq[0] * (x - c * t)) * np.sin(np.pi * freq[1] * (y - c * t)) 

114 

115 elif ndim == 3: 

116 x, y, z = self.grids 

117 sol[:] = ( 

118 np.sin(np.pi * freq[0] * (x - c * t)) 

119 * np.sin(np.pi * freq[1] * (y - c * t)) 

120 * np.sin(np.pi * freq[2] * (z - c * t)) 

121 ) 

122 

123 return sol
