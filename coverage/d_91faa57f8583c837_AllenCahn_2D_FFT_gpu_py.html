9class allencahn2d_imex(ptype): # pragma: no cover 

10 """ 

11 Example implementing Allen-Cahn equation in 2D using FFTs for solving linear parts, IMEX time-stepping 

12 

13 Parameters 

14 ---------- 

15 nvars : int 

16 Number of unknowns in the problem. 

17 nu : float 

18 Problem parameter. 

19 eps : float 

20 Problem parameter. 

21 radius : float 

22 Radius of the circles. 

23 L : int 

24 Denotes the period of the function to be approximated for the Fourier transform. 

25 init_type : str 

26 Indicates which type of initial condition is used. 

27 

28 Attributes 

29 ---------- 

30 xvalues : cp.ndarray 

31 Grid points in space. 

32 dx : float 

33 Cupy mesh width. 

34 lap : cp.ndarray 

35 Spectral operator for Laplacian. 

36 rfft_object : 

37 Planned real FFT for forward transformation. 

38 irfft_object : 

39 Planned IFFT for backward transformation. 

40 """ 

41 

42 dtype_u = cupy_mesh 

43 dtype_f = imex_cupy_mesh

45 def __init__(self, nvars=None, nu=2, eps=0.04, radius=0.25, L=1.0, init_type='circle'): 

46 """Initialization routine""" 

47 

48 if nvars is None: 

49 nvars = [(256, 256), (64, 64)] 

50 

51 # we assert that nvars looks very particular here.. this will be necessary for coarsening in space later on 

52 if len(nvars) != 2: 

53 raise ProblemError('this is a 2d example, got %s' % nvars) 

54 if nvars[0] != nvars[1]: 

55 raise ProblemError('need a square domain, got %s' % nvars) 

56 if nvars[0] % 2 != 0: 

57 raise ProblemError('the setup requires nvars = 2^p per dimension') 

58 

59 # invoke super init, passing number of dofs, dtype_u and dtype_f 

60 super().__init__(init=(nvars, None, cp.dtype('float64'))) 

61 self._makeAttributeAndRegister( 

62 'nvars', 'nu', 'eps', 'radius', 'L', 'init_type', localVars=locals(), readOnly=True 

63 ) 

64 

65 self.dx = self.L / self.nvars[0] # could be useful for hooks, too. 

66 self.xvalues = cp.array([i * self.dx - self.L / 2.0 for i in range(self.nvars[0])]) 

67 

68 kx = cp.zeros(self.init[0][0]) 

69 ky = cp.zeros(self.init[0][1] // 2 + 1) 

70 

71 kx[: int(self.init[0][0] / 2) + 1] = 2 * np.pi / self.L * cp.arange(0, int(self.init[0][0] / 2) + 1) 

72 kx[int(self.init[0][0] / 2) + 1 :] = ( 

73 2 * np.pi / self.L * cp.arange(int(self.init[0][0] / 2) + 1 - self.init[0][0], 0) 

74 ) 

75 ky[:] = 2 * np.pi / self.L * cp.arange(0, self.init[0][1] // 2 + 1) 

76 

77 xv, yv = cp.meshgrid(kx, ky, indexing='ij') 

78 self.lap = -(xv**2) - yv**2 

79 

80 def eval_f(self, u, t): 

81 """ 

82 Routine to evaluate the right-hand side of the problem. 

83 

84 Parameters 

85 ---------- 

86 u : dtype_u 

87 Current values of the numerical solution. 

88 t : float 

89 Current time of the numerical solution is computed. 

90 

91 Returns 

92 ------- 

93 f : dtype_f 

94 The right-hand side of the problem. 

95 """ 

96 

97 f = self.dtype_f(self.init) 

98 v = u.flatten() 

99 tmp = self.lap * cp.fft.rfft2(u) 

100 f.impl[:] = cp.fft.irfft2(tmp) 

101 if self.eps > 0: 

102 f.expl[:] = (1.0 / self.eps**2 * v * (1.0 - v**self.nu)).reshape(self.nvars) 

103 return f

105 def solve_system(self, rhs, factor, u0, t): 

106 """ 

107 Simple FFT solver for the diffusion part. 

108 

109 Parameters 

110 ---------- 

111 rhs : dtype_f 

112 Right-hand side for the linear system. 

113 factor : float 

114 Abbrev. for the node-to-node stepsize (or any other factor required). 

115 u0 : dtype_u 

116 Initial guess for the iterative solver (not used here so far). 

117 t : float 

118 Current time (e.g. for time-dependent BCs). 

119 

120 Parameters 

121 ---------- 

122 me : dtype_u 

123 The solution as mesh. 

124 """ 

125 

126 me = self.dtype_u(self.init) 

127 tmp = cp.fft.rfft2(rhs) / (1.0 - factor * self.lap) 

128 me[:] = cp.fft.irfft2(tmp) 

129 

130 return me

132 def u_exact(self, t): 

133 """ 

134 Routine to compute the exact solution at time t. 

135 

136 Parameters 

137 ---------- 

138 t : float 

139 Time of the exact solution. 

140 

141 Returns 

142 ------- 

143 me : dtype_u 

144 The exact solution. 

145 """ 

146 

147 assert t == 0, 'ERROR: u_exact only valid for t=0' 

148 me = self.dtype_u(self.init, val=0.0) 

149 if self.init_type == 'circle': 

150 xv, yv = cp.meshgrid(self.xvalues, self.xvalues, indexing='ij') 

151 me[:, :] = cp.tanh((self.radius - cp.sqrt(xv**2 + yv**2)) / (cp.sqrt(2) * self.eps)) 

152 elif self.init_type == 'checkerboard': 

153 xv, yv = cp.meshgrid(self.xvalues, self.xvalues) 

154 me[:, :] = cp.sin(2.0 * np.pi * xv) * cp.sin(2.0 * np.pi * yv) 

155 elif self.init_type == 'random': 

156 me[:, :] = cp.random.uniform(-1, 1, self.init) 

157 else: 

158 raise NotImplementedError('type of initial value not implemented, got %s' % self.init_type) 

159 

160 return me
