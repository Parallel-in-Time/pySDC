17 def __init__(self, params): 

18 """ 

19 Initialization routine for the custom sweeper 

20 

21 Args: 

22 params: parameters for the sweeper 

23 """ 

24 

25 if 'QI' not in params: 

26 params['QI'] = 'IE' 

27 if 'QE' not in params: 

28 params['QE'] = 'EE' 

29 

30 # call parent's initialization routine 

31 super(imex_1st_order, self).__init__(params) 

32 

33 # IMEX integration matrices 

34 self.QI = self.get_Qdelta_implicit(coll=self.coll, qd_type=self.params.QI) 

35 self.QE = self.get_Qdelta_explicit(coll=self.coll, qd_type=self.params.QE)

37 def integrate(self): 

38 """ 

39 Integrates the right-hand side (here impl + expl) 

40 

41 Returns: 

42 list of dtype_u: containing the integral as values 

43 """ 

44 

45 # get current level and problem description 

46 L = self.level 

47 

48 me = [] 

49 

50 # integrate RHS over all collocation nodes 

51 for m in range(1, self.coll.num_nodes + 1): 

52 me.append(L.dt * self.coll.Qmat[m, 1] * (L.f[1].impl + L.f[1].expl)) 

53 # new instance of dtype_u, initialize values with 0 

54 for j in range(2, self.coll.num_nodes + 1): 

55 me[m - 1] += L.dt * self.coll.Qmat[m, j] * (L.f[j].impl + L.f[j].expl) 

56 

57 return me

59 def update_nodes(self): 

60 """ 

61 Update the u- and f-values at the collocation nodes -> corresponds to a single sweep over all nodes 

62 

63 Returns: 

64 None 

65 """ 

66 

67 # get current level and problem description 

68 L = self.level 

69 P = L.prob 

70 

71 # only if the level has been touched before 

72 assert L.status.unlocked 

73 

74 # get number of collocation nodes for easier access 

75 M = self.coll.num_nodes 

76 

77 # gather all terms which are known already (e.g. from the previous iteration) 

78 # this corresponds to u0 + QF(u^k) - QIFI(u^k) - QEFE(u^k) + tau 

79 

80 # get QF(u^k) 

81 integral = self.integrate() 

82 for m in range(M): 

83 # subtract QIFI(u^k)_m + QEFE(u^k)_m 

84 for j in range(1, M + 1): 

85 integral[m] -= L.dt * (self.QI[m + 1, j] * L.f[j].impl + self.QE[m + 1, j] * L.f[j].expl) 

86 # add initial value 

87 integral[m] += L.u[0] 

88 # add tau if associated 

89 if L.tau[m] is not None: 

90 integral[m] += L.tau[m] 

91 

92 # do the sweep 

93 for m in range(0, M): 

94 # build rhs, consisting of the known values from above and new values from previous nodes (at k+1) 

95 rhs = P.dtype_u(integral[m]) 

96 for j in range(1, m + 1): 

97 rhs += L.dt * (self.QI[m + 1, j] * L.f[j].impl + self.QE[m + 1, j] * L.f[j].expl) 

98 

99 # implicit solve with prefactor stemming from QI 

100 L.u[m + 1] = P.solve_system( 

101 rhs, L.dt * self.QI[m + 1, m + 1], L.u[m + 1], L.time + L.dt * self.coll.nodes[m] 

102 ) 

103 

104 # update function values 

105 L.f[m + 1] = P.eval_f(L.u[m + 1], L.time + L.dt * self.coll.nodes[m]) 

106 

107 # indicate presence of new values at this level 

108 L.status.updated = True 

109 

110 return None

112 def compute_end_point(self): 

113 """ 

114 Compute u at the right point of the interval 

115 

116 The value uend computed here is a full evaluation of the Picard formulation unless do_full_update==False 

117 

118 Returns: 

119 None 

120 """ 

121 

122 # get current level and problem description 

123 L = self.level 

124 P = L.prob 

125 

126 # check if Mth node is equal to right point and do_coll_update is false, perform a simple copy 

127 if self.coll.right_is_node and not self.params.do_coll_update: 

128 # a copy is sufficient 

129 L.uend = P.dtype_u(L.u[-1]) 

130 else: 

131 # start with u0 and add integral over the full interval (using coll.weights) 

132 L.uend = P.dtype_u(L.u[0]) 

133 for m in range(self.coll.num_nodes): 

134 L.uend += L.dt * self.coll.weights[m] * (L.f[m + 1].impl + L.f[m + 1].expl) 

135 # add up tau correction of the full interval (last entry) 

136 if L.tau[-1] is not None: 

137 L.uend += L.tau[-1] 

138 

139 return None

141 def get_sweeper_mats(self): 

142 """ 

143 Returns the three matrices Q, QI, QE which define the sweeper. 

144 

145 The first row and column, corresponding to the left starting value, are removed to correspond to the notation 

146 introduced in Ruprecht & Speck, Spectral deferred corrections with fast-wave slow-wave splitting, 2016 

147 """ 

148 QE = self.QE[1:, 1:] 

149 QI = self.QI[1:, 1:] 

150 Q = self.coll.Qmat[1:, 1:] 

151 return QE, QI, Q

153 def get_scalar_problems_sweeper_mats(self, lambdas=None): 

154 """ 

155 This function returns the corresponding matrices of an IMEX-SDC sweep in matrix formulation. 

156 

157 See Ruprecht & Speck, Spectral deferred corrections with fast-wave slow-wave splitting, 2016 for the derivation. 

158 

159 Args: 

160 lambdas (numpy.ndarray): the first entry in lambdas is lambda_fast, the second is lambda_slow. 

161 """ 

162 QE, QI, Q = self.get_sweeper_mats() 

163 if lambdas is None: 

164 pass 

165 # should use lambdas from attached problem and make sure it is a scalar IMEX 

166 raise NotImplementedError("At the moment, the values for lambda have to be provided") 

167 else: 

168 lambda_fast = lambdas[0] 

169 lambda_slow = lambdas[1] 

170 nnodes = self.coll.num_nodes 

171 dt = self.level.dt 

172 LHS = np.eye(nnodes) - dt * (lambda_fast * QI + lambda_slow * QE) 

173 RHS = dt * ((lambda_fast + lambda_slow) * Q - (lambda_fast * QI + lambda_slow * QE)) 

174 return LHS, RHS

176 def get_scalar_problems_manysweep_mat(self, nsweeps, lambdas=None): 

177 """ 

178 For a scalar problem, K sweeps of IMEX-SDC can be written in matrix form. 

179 

180 Args: 

181 nsweeps (int): number of sweeps 

182 lambdas (numpy.ndarray): the first entry in lambdas is lambda_fast, the second is lambda_slow. 

183 """ 

184 LHS, RHS = self.get_scalar_problems_sweeper_mats(lambdas=lambdas) 

185 Pinv = np.linalg.inv(LHS) 

186 Mat_sweep = np.linalg.matrix_power(Pinv.dot(RHS), nsweeps) 

187 for k in range(0, nsweeps): 

188 Mat_sweep += np.linalg.matrix_power(Pinv.dot(RHS), k).dot(Pinv) 

189 return Mat_sweep
