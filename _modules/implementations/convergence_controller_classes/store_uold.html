from pySDC.core.ConvergenceController import ConvergenceController



[docs]class StoreUOld(ConvergenceController):
    """
    Class to store the solution of the last iteration in a variable called 'uold' of the levels.

    Default control order is 90.
    """


[docs]    def setup(self, controller, params, description, **kwargs):

        Define parameters here

            (dict): The updated params dictionary

[docs]    def post_iteration_processing(self, controller, S, **kwargs):
        """
        Store the solution at the current iteration

        Args:
            controller (pySDC.Controller): The controller
            S (pySDC.Step): The current step

        Return:
            None
        """
        for L in S.levels:
            L.uold[:] = L.u[:]

        return None
