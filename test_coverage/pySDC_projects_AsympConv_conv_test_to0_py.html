12def compute_and_plot_specrad(Nnodes, lam): 

13 """ 

14 Compute and plot the spectral radius of the smoother for different step-sizes 

15 

16 Args: 

17 Nnodes: number of collocation nodes 

18 lam: test parameter representing the spatial problem 

19 """ 

20 

21 coll = CollGaussRadau_Right(Nnodes, 0, 1) 

22 Qmat = coll.Qmat[1:, 1:] 

23 

24 # do LU decomposition of QT (St. Martin's trick) 

25 QT = coll.Qmat[1:, 1:].T 

26 [_, _, U] = LA.lu(QT, overwrite_a=True) 

27 QDmat = U.T 

28 

29 Nmat = np.zeros((Nnodes, Nnodes)) 

30 Nmat[:, -1] = 1 

31 

32 Nsteps_list = [64, 256] 

33 color_list = ['red', 'blue'] 

34 marker_list = ['s', 'o'] 

35 

36 setup_list = zip(Nsteps_list, color_list, marker_list) 

37 

38 xlist = [0.1 ** i for i in range(11)] 

39 

40 rc('font', **{"sans-serif": ["Arial"], "size": 24}) 

41 plt.subplots(figsize=(15, 10)) 

42 

43 for Nsteps, color, marker in setup_list: 

44 

45 Emat = np.zeros((Nsteps, Nsteps)) 

46 np.fill_diagonal(Emat[1:, :], 1) 

47 

48 Prho_list = [] 

49 predict_list = [] 

50 for x in xlist: 

51 

52 mat = np.linalg.inv(np.eye(Nnodes * Nsteps) - x * lam * np.kron(np.eye(Nsteps), QDmat)).dot( 

53 x * lam * np.kron(np.eye(Nsteps), (Qmat - QDmat)) + np.kron(Emat, Nmat)) 

54 

55 Prho_list.append(max(abs(np.linalg.eigvals(mat)))) 

56 # predict_list.append((1 + x) ** (1.0 - 1.0 / (Nnodes * Nsteps)) * x ** (1.0 / (Nnodes * Nsteps))) 

57 predict_list.append(x ** (1.0 / (Nsteps))) 

58 

59 if len(predict_list) > 1: 

60 print(x, predict_list[-1], Prho_list[-1], Prho_list[-2] / Prho_list[-1], 

61 predict_list[-2] / predict_list[-1]) 

62 

63 plt.loglog(xlist, Prho_list, linestyle='-', linewidth=3, color=color, marker=marker, markersize=10, 

64 label='spectral radius, L=' + str(Nsteps)) 

65 plt.loglog(xlist, predict_list, linestyle='--', linewidth=2, color=color, marker=marker, 

66 markersize=10, label='estimate, L=' + str(Nsteps)) 

67 

68 ax = plt.gca() 

69 ax.invert_xaxis() 

70 

71 plt.xlabel('time-step size') 

72 plt.ylabel('spectral radius') 

73 plt.legend(loc=3, numpoints=1) 

74 plt.grid() 

75 plt.ylim([1E-02, 1E01])
