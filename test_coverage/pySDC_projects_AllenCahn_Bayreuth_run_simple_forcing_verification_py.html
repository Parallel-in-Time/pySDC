20def run_simulation(name='', spectral=None, nprocs_space=None): 

21 """ 

22 A test program to do PFASST runs for the AC equation with different forcing 

23 

24 Args: 

25 name (str): name of the run, will be used to distinguish different setups 

26 spectral (bool): run in real or spectral space 

27 nprocs_space (int): number of processors in space (None if serial) 

28 """ 

29 

30 # set MPI communicator 

31 comm = MPI.COMM_WORLD 

32 

33 world_rank = comm.Get_rank() 

34 world_size = comm.Get_size() 

35 

36 # split world communicator to create space-communicators 

37 if nprocs_space is not None: 

38 color = int(world_rank / nprocs_space) 

39 else: 

40 color = int(world_rank / 1) 

41 space_comm = comm.Split(color=color) 

42 space_rank = space_comm.Get_rank() 

43 space_size = space_comm.Get_size() 

44 

45 assert world_size == space_size, 'This script cannot run parallel-in-time with MPI, only spatial parallelism' 

46 

47 # initialize level parameters 

48 level_params = dict() 

49 level_params['restol'] = 1E-08 

50 level_params['dt'] = 1E-03 

51 level_params['nsweeps'] = [1] 

52 

53 # initialize sweeper parameters 

54 sweeper_params = dict() 

55 sweeper_params['collocation_class'] = CollGaussRadau_Right 

56 sweeper_params['num_nodes'] = [3] 

57 sweeper_params['QI'] = ['LU'] # For the IMEX sweeper, the LU-trick can be activated for the implicit part 

58 sweeper_params['initial_guess'] = 'spread' 

59 

60 # initialize problem parameters 

61 problem_params = dict() 

62 problem_params['L'] = 1.0 

63 problem_params['nvars'] = [(128, 128), (32, 32)] 

64 problem_params['eps'] = [0.04] 

65 problem_params['radius'] = 0.25 

66 problem_params['comm'] = space_comm 

67 problem_params['name'] = name 

68 problem_params['init_type'] = 'circle' 

69 problem_params['spectral'] = spectral 

70 

71 if name == 'AC-test-constforce': 

72 problem_params['dw'] = [-23.59] 

73 

74 # initialize step parameters 

75 step_params = dict() 

76 step_params['maxiter'] = 50 

77 

78 # initialize controller parameters 

79 controller_params = dict() 

80 controller_params['logger_level'] = 30 if space_rank == 0 else 99 # set level depending on rank 

81 controller_params['hook_class'] = monitor 

82 controller_params['predict_type'] = 'pfasst_burnin' 

83 

84 # fill description dictionary for easy step instantiation 

85 description = dict() 

86 description['problem_params'] = problem_params # pass problem parameters 

87 description['sweeper_class'] = imex_1st_order 

88 description['sweeper_params'] = sweeper_params # pass sweeper parameters 

89 description['level_params'] = level_params # pass level parameters 

90 description['step_params'] = step_params # pass step parameters 

91 description['space_transfer_class'] = fft_to_fft 

92 

93 if name == 'AC-test-noforce' or name == 'AC-test-constforce': 

94 description['problem_class'] = allencahn_imex 

95 elif name == 'AC-test-timeforce': 

96 description['problem_class'] = allencahn_imex_timeforcing 

97 else: 

98 raise NotImplementedError(f'{name} is not implemented') 

99 

100 # set time parameters 

101 t0 = 0.0 

102 Tend = 32 * 0.001 

103 

104 if space_rank == 0: 

105 out = f'---------> Running {name} with spectral={spectral} and {space_size} process(es) in space...' 

106 print(out) 

107 

108 # instantiate controller 

109 controller = controller_nonMPI(num_procs=8, controller_params=controller_params, description=description) 

110 

111 # get initial values on finest level 

112 P = controller.MS[0].levels[0].prob 

113 uinit = P.u_exact(t0) 

114 

115 # call main function to get things done... 

116 uend, stats = controller.run(u0=uinit, t0=t0, Tend=Tend) 

117 

118 if space_rank == 0: 

119 

120 # convert filtered statistics to list of computed radii, sorted by time 

121 computed_radii = sort_stats(filter_stats(stats, type='computed_radius'), sortby='time') 

122 exact_radii = sort_stats(filter_stats(stats, type='exact_radius'), sortby='time') 

123 computed_vol = sort_stats(filter_stats(stats, type='computed_volume'), sortby='time') 

124 exact_vol = sort_stats(filter_stats(stats, type='exact_volume'), sortby='time') 

125 

126 # print and store radii and error over time 

127 err_test = 0.0 

128 results = dict() 

129 for cr, er, cv, ev in zip(computed_radii, exact_radii, computed_vol, exact_vol): 

130 if name == 'AC-test-noforce': 

131 exrad = er[1] 

132 exvol = ev[1] 

133 else: 

134 exrad = computed_radii[0][1] 

135 exvol = computed_vol[0][1] 

136 if exrad > 0: 

137 errr = abs(cr[1] - exrad) / exrad 

138 errv = abs(cv[1] - exvol) / exvol 

139 else: 

140 errr = 1.0 

141 errv = 1.0 

142 if cr[0] == 0.025: 

143 err_test = errr 

144 out = f'Computed/exact/error radius for time {cr[0]:6.4f}: ' \ 

145 f'{cr[1]:8.6f} / {exrad:8.6f} / {errr:6.4e}' 

146 print(out) 

147 results[cr[0]] = (cr[1], exrad, errr, cv[1], exvol, errv) 

148 fname = f'./data/{name}_results.json' 

149 with open(fname, 'w') as fp: 

150 json.dump(results, fp, sort_keys=True, indent=4) 

151 

152 print() 

153 

154 # convert filtered statistics of iterations count, sorted by time 

155 iter_counts = sort_stats(filter_stats(stats, type='niter'), sortby='time') 

156 niters = np.mean(np.array([item[1] for item in iter_counts])) 

157 out = f'Mean number of iterations: {niters:.4f}' 

158 print(out) 

159 

160 # get setup time 

161 timing = sort_stats(filter_stats(stats, type='timing_setup'), sortby='time') 

162 out = f'Setup time: {timing[0][1]:.4f} sec.' 

163 print(out) 

164 

165 # get running time 

166 timing = sort_stats(filter_stats(stats, type='timing_run'), sortby='time') 

167 out = f'Time to solution: {timing[0][1]:.4f} sec.' 

168 print(out) 

169 

170 out = '...Done <---------\n' 

171 print(out) 

172 

173 # Testing the output 

174 if name == 'AC-test-noforce': 

175 if spectral: 

176 exp_iters = 7.0 

177 exp_err = 7.821e-02 

178 else: 

179 exp_iters = 8.25 

180 exp_err = 7.85e-02 

181 elif name == 'AC-test-constforce': 

182 if spectral: 

183 exp_iters = 3.90625 

184 exp_err = 4.678e-04 

185 else: 

186 exp_iters = 5.21875 

187 exp_err = 6.2384e-04 

188 elif name == 'AC-test-timeforce': 

189 if spectral: 

190 exp_iters = 2.03125 

191 exp_err = 6.2345e-04 

192 else: 

193 exp_iters = 3.15625 

194 exp_err = 6.2345e-04 

195 else: 

196 raise NotImplementedError(f'{name} is not implemented') 

197 

198 assert niters == exp_iters, f'Got deviating iteration counts of {niters} instead of {exp_iters}' 

199 assert err_test < exp_err, f'Got deviating errors of {err_test} instead of {exp_err}'
