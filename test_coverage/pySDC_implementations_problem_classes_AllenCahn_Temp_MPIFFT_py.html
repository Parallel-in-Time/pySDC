11class allencahn_temp_imex(ptype): 

12 """ 

13 Example implementing Allen-Cahn equation in 2-3D using mpi4py-fft for solving linear parts, IMEX time-stepping 

14 

15 mpi4py-fft: https://mpi4py-fft.readthedocs.io/en/latest/ 

16 

17 Attributes: 

18 fft: fft object 

19 X: grid coordinates in real space 

20 K2: Laplace operator in spectral space 

21 dx: mesh width in x direction 

22 dy: mesh width in y direction 

23 """

25 def __init__(self, problem_params, dtype_u=mesh, dtype_f=imex_mesh): 

26 """ 

27 Initialization routine 

28 

29 Args: 

30 problem_params (dict): custom parameters for the example 

31 dtype_u: fft data type (will be passed to parent class) 

32 dtype_f: fft data type wuth implicit and explicit parts (will be passed to parent class) 

33 """ 

34 

35 if 'L' not in problem_params: 

36 problem_params['L'] = 1.0 

37 if 'init_type' not in problem_params: 

38 problem_params['init_type'] = 'circle' 

39 if 'comm' not in problem_params: 

40 problem_params['comm'] = None 

41 if 'dw' not in problem_params: 

42 problem_params['dw'] = 1.0 

43 

44 # these parameters will be used later, so assert their existence 

45 essential_keys = ['nvars', 'eps', 'L', 'radius', 'dw', 'spectral', 'TM', 'D'] 

46 for key in essential_keys: 

47 if key not in problem_params: 

48 msg = 'need %s to instantiate problem, only got %s' % (key, str(problem_params.keys())) 

49 raise ParameterError(msg) 

50 

51 if not (isinstance(problem_params['nvars'], tuple) and len(problem_params['nvars']) > 1): 

52 raise ProblemError('Need at least two dimensions') 

53 

54 # creating FFT structure 

55 ndim = len(problem_params['nvars']) 

56 axes = tuple(range(ndim)) 

57 self.fft = PFFT(problem_params['comm'], list(problem_params['nvars']), axes=axes, dtype=np.float64, 

58 collapse=True) 

59 

60 # get test data to figure out type and dimensions 

61 tmp_u = newDistArray(self.fft, problem_params['spectral']) 

62 

63 # add two components to contain field and temperature 

64 self.ncomp = 2 

65 sizes = tmp_u.shape + (self.ncomp,) 

66 

67 # invoke super init, passing the communicator and the local dimensions as init 

68 super(allencahn_temp_imex, self).__init__(init=(sizes, problem_params['comm'], tmp_u.dtype), 

69 dtype_u=dtype_u, dtype_f=dtype_f, params=problem_params) 

70 

71 L = np.array([self.params.L] * ndim, dtype=float) 

72 

73 # get local mesh 

74 X = np.ogrid[self.fft.local_slice(False)] 

75 N = self.fft.global_shape() 

76 for i in range(len(N)): 

77 X[i] = (X[i] * L[i] / N[i]) 

78 self.X = [np.broadcast_to(x, self.fft.shape(False)) for x in X] 

79 

80 # get local wavenumbers and Laplace operator 

81 s = self.fft.local_slice() 

82 N = self.fft.global_shape() 

83 k = [np.fft.fftfreq(n, 1. / n).astype(int) for n in N[:-1]] 

84 k.append(np.fft.rfftfreq(N[-1], 1. / N[-1]).astype(int)) 

85 K = [ki[si] for ki, si in zip(k, s)] 

86 Ks = np.meshgrid(*K, indexing='ij', sparse=True) 

87 Lp = 2 * np.pi / L 

88 for i in range(ndim): 

89 Ks[i] = (Ks[i] * Lp[i]).astype(float) 

90 K = [np.broadcast_to(k, self.fft.shape(True)) for k in Ks] 

91 K = np.array(K).astype(float) 

92 self.K2 = np.sum(K * K, 0, dtype=float) 

93 

94 # Need this for diagnostics 

95 self.dx = self.params.L / problem_params['nvars'][0] 

96 self.dy = self.params.L / problem_params['nvars'][1]

98 def eval_f(self, u, t): 

99 """ 

100 Routine to evaluate the RHS 

101 

102 Args: 

103 u (dtype_u): current values 

104 t (float): current time 

105 

106 Returns: 

107 dtype_f: the RHS 

108 """ 

109 

110 f = self.dtype_f(self.init) 

111 

112 if self.params.spectral: 

113 

114 f.impl[..., 0] = -self.K2 * u[..., 0] 

115 

116 if self.params.eps > 0: 

117 tmp_u = newDistArray(self.fft, False) 

118 tmp_T = newDistArray(self.fft, False) 

119 tmp_u = self.fft.backward(u[..., 0], tmp_u) 

120 tmp_T = self.fft.backward(u[..., 1], tmp_T) 

121 tmpf = - 2.0 / self.params.eps ** 2 * tmp_u * (1.0 - tmp_u) * (1.0 - 2.0 * tmp_u) - \ 

122 6.0 * self.params.dw * (tmp_T - self.params.TM) / self.params.TM * tmp_u * (1.0 - tmp_u) 

123 f.expl[..., 0] = self.fft.forward(tmpf) 

124 

125 f.impl[..., 1] = -self.params.D * self.K2 * u[..., 1] 

126 f.expl[..., 1] = f.impl[..., 0] + f.expl[..., 0] 

127 

128 else: 

129 

130 u_hat = self.fft.forward(u[..., 0]) 

131 lap_u_hat = -self.K2 * u_hat 

132 f.impl[..., 0] = self.fft.backward(lap_u_hat, f.impl[..., 0]) 

133 

134 if self.params.eps > 0: 

135 f.expl[..., 0] = -2.0 / self.params.eps ** 2 * u[..., 0] * (1.0 - u[..., 0]) * (1.0 - 2.0 * u[..., 0]) 

136 f.expl[..., 0] -= 6.0 * self.params.dw * (u[..., 1] - self.params.TM) / self.params.TM * u[..., 0] * \ 

137 (1.0 - u[..., 0]) 

138 

139 u_hat = self.fft.forward(u[..., 1]) 

140 lap_u_hat = -self.params.D * self.K2 * u_hat 

141 f.impl[..., 1] = self.fft.backward(lap_u_hat, f.impl[..., 1]) 

142 f.expl[..., 1] = f.impl[..., 0] + f.expl[..., 0] 

143 

144 return f

146 def solve_system(self, rhs, factor, u0, t): 

147 """ 

148 Simple FFT solver for the diffusion part 

149 

150 Args: 

151 rhs (dtype_f): right-hand side for the linear system 

152 factor (float) : abbrev. for the node-to-node stepsize (or any other factor required) 

153 u0 (dtype_u): initial guess for the iterative solver (not used here so far) 

154 t (float): current time (e.g. for time-dependent BCs) 

155 

156 Returns: 

157 dtype_u: solution as mesh 

158 """ 

159 

160 if self.params.spectral: 

161 

162 me = self.dtype_u(self.init) 

163 me[..., 0] = rhs[..., 0] / (1.0 + factor * self.K2) 

164 me[..., 1] = rhs[..., 1] / (1.0 + factor * self.params.D * self.K2) 

165 

166 else: 

167 

168 me = self.dtype_u(self.init) 

169 rhs_hat = self.fft.forward(rhs[..., 0]) 

170 rhs_hat /= (1.0 + factor * self.K2) 

171 me[..., 0] = self.fft.backward(rhs_hat) 

172 rhs_hat = self.fft.forward(rhs[..., 1]) 

173 rhs_hat /= (1.0 + factor * self.params.D * self.K2) 

174 me[..., 1] = self.fft.backward(rhs_hat) 

175 

176 return me

178 def u_exact(self, t): 

179 """ 

180 Routine to compute the exact solution at time t 

181 

182 Args: 

183 t (float): current time 

184 

185 Returns: 

186 dtype_u: exact solution 

187 """

189 def circle(): 

190 tmp_me = newDistArray(self.fft, self.params.spectral) 

191 r2 = (self.X[0] - 0.5) ** 2 + (self.X[1] - 0.5) ** 2 

192 if self.params.spectral: 

193 tmp = 0.5 * (1.0 + np.tanh((self.params.radius - np.sqrt(r2)) / (np.sqrt(2) * self.params.eps))) 

194 tmp_me[:] = self.fft.forward(tmp) 

195 else: 

196 tmp_me[:] = 0.5 * (1.0 + np.tanh((self.params.radius - np.sqrt(r2)) / (np.sqrt(2) * self.params.eps))) 

197 return tmp_me 

198 

199 def circle_rand(): 

200 tmp_me = newDistArray(self.fft, self.params.spectral) 

201 ndim = len(tmp_me.shape) 

202 L = int(self.params.L) 

203 # get random radii for circles/spheres 

204 np.random.seed(1) 

205 lbound = 3.0 * self.params.eps 

206 ubound = 0.5 - self.params.eps 

207 rand_radii = (ubound - lbound) * np.random.random_sample(size=tuple([L] * ndim)) + lbound 

208 # distribute circles/spheres 

209 tmp = newDistArray(self.fft, False) 

210 if ndim == 2: 

211 for i in range(0, L): 

212 for j in range(0, L): 

213 # build radius 

214 r2 = (self.X[0] + i - L + 0.5) ** 2 + (self.X[1] + j - L + 0.5) ** 2 

215 # add this blob, shifted by 1 to avoid issues with adding up negative contributions 

216 tmp += np.tanh((rand_radii[i, j] - np.sqrt(r2)) / (np.sqrt(2) * self.params.eps)) + 1 

217 # normalize to [0,1] 

218 tmp *= 0.5 

219 assert np.all(tmp <= 1.0) 

220 if self.params.spectral: 

221 tmp_me[:] = self.fft.forward(tmp) 

222 else: 

223 tmp_me[:] = tmp[:] 

224 return tmp_me 

225 

226 def sine(): 

227 tmp_me = newDistArray(self.fft, self.params.spectral) 

228 if self.params.spectral: 

229 tmp = 0.5 * (2.0 + 0.0 * np.sin(2 * np.pi * self.X[0]) * np.sin(2 * np.pi * self.X[1])) 

230 tmp_me[:] = self.fft.forward(tmp) 

231 else: 

232 tmp_me[:] = 0.5 * (2.0 + 0.0 * np.sin(2 * np.pi * self.X[0]) * np.sin(2 * np.pi * self.X[1])) 

233 return tmp_me 

234 

235 assert t == 0, 'ERROR: u_exact only valid for t=0' 

236 me = self.dtype_u(self.init, val=0.0) 

237 if self.params.init_type == 'circle': 

238 me[..., 0] = circle() 

239 me[..., 1] = sine() 

240 elif self.params.init_type == 'circle_rand': 

241 me[..., 0] = circle_rand() 

242 me[..., 1] = sine() 

243 else: 

244 raise NotImplementedError('type of initial value not implemented, got %s' % self.params.init_type) 

245 

246 return me
