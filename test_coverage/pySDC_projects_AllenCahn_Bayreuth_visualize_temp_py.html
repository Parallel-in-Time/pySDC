13def plot_data(path='./data', name='', output='.'): 

14 """ 

15 Visualization using numpy arrays (written via MPI I/O) and json description

17 Produces one png file per time-step, combine as movie via e.g. 

18 > ffmpeg -i data/name_%08d.png name.mp4 

19 

20 Args: 

21 path (str): path to data files 

22 name (str): name of the simulation (expects data to be in data path) 

23 output (str): path to output

25 

26 json_files = sorted(glob.glob(f'{path}/{name}_*.json'))

34 print(f'Working on step {index}...')

36 array = np.fromfile(data_file, dtype=obj['datatype'])
