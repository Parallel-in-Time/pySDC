18def findomega(stab_fh): 

19 assert np.array_equal(np.shape(stab_fh), [2, 2]), 'Not 2x2 matrix...' 

20 omega = sympy.Symbol('omega') 

21 func = (sympy.exp(-1j * omega) - stab_fh[0, 0]) * (sympy.exp(-1j * omega) - stab_fh[1, 1]) - \ 

22 stab_fh[0, 1] * stab_fh[1, 0] 

23 solsym = sympy.solve(func, omega) 

24 sol0 = complex(solsym[0]) 

25 sol1 = complex(solsym[1]) 

26 if sol0.real >= 0: 

27 sol = sol0 

28 elif sol1.real >= 0: 

29 sol = sol1 

30 else: 

31 print("Two roots with real part of same sign...") 

32 sol = sol0 

33 return sol

36def compute_and_plot_dispersion(): 

37 problem_params = dict() 

38 # SET VALUE FOR lambda_slow AND VALUES FOR lambda_fast ### 

39 problem_params['lambda_s'] = np.array([0.0]) 

40 problem_params['lambda_f'] = np.array([0.0]) 

41 problem_params['u0'] = 1.0 

42 

43 # initialize sweeper parameters 

44 sweeper_params = dict() 

45 # SET TYPE AND NUMBER OF QUADRATURE NODES ### 

46 sweeper_params['collocation_class'] = CollGaussRadau_Right 

47 sweeper_params['do_coll_update'] = True 

48 sweeper_params['num_nodes'] = 3 

49 

50 # initialize level parameters 

51 level_params = dict() 

52 level_params['dt'] = 1.0 

53 

54 # fill description dictionary for easy step instantiation 

55 description = dict() 

56 description['problem_class'] = swfw_scalar # pass problem class 

57 description['problem_params'] = problem_params # pass problem parameters 

58 description['sweeper_class'] = imex_1st_order # pass sweeper 

59 description['sweeper_params'] = sweeper_params # pass sweeper parameters 

60 description['level_params'] = level_params # pass level parameters 

61 description['step_params'] = dict() # pass step parameters 

62 

63 # SET NUMBER OF ITERATIONS ### 

64 K = 3 

65 

66 # ORDER OF DIRK/IMEX IS EQUAL TO NUMBER OF ITERATIONS AND THUS ORDER OF SDC ### 

67 dirk_order = K 

68 

69 c_speed = 1.0 

70 U_speed = 0.05 

71 

72 # now the description contains more or less everything we need to create a step 

73 S = step(description=description) 

74 

75 L = S.levels[0] 

76 

77 # u0 = S.levels[0].prob.u_exact(t0) 

78 # S.init_step(u0) 

79 QE = L.sweep.QE[1:, 1:] 

80 QI = L.sweep.QI[1:, 1:] 

81 Q = L.sweep.coll.Qmat[1:, 1:] 

82 nnodes = L.sweep.coll.num_nodes 

83 dt = L.params.dt 

84 

85 Nsamples = 15 

86 k_vec = np.linspace(0, np.pi, Nsamples + 1, endpoint=False) 

87 k_vec = k_vec[1:] 

88 phase = np.zeros((3, Nsamples)) 

89 amp_factor = np.zeros((3, Nsamples)) 

90 

91 for i in range(0, np.size(k_vec)): 

92 

93 Cs = -1j * k_vec[i] * np.array([[0.0, c_speed], [c_speed, 0.0]], dtype='complex') 

94 Uadv = -1j * k_vec[i] * np.array([[U_speed, 0.0], [0.0, U_speed]], dtype='complex') 

95 

96 LHS = np.eye(2 * nnodes) - dt * (np.kron(QI, Cs) + np.kron(QE, Uadv)) 

97 RHS = dt * (np.kron(Q, Uadv + Cs) - np.kron(QI, Cs) - np.kron(QE, Uadv)) 

98 

99 LHSinv = np.linalg.inv(LHS) 

100 Mat_sweep = np.linalg.matrix_power(LHSinv.dot(RHS), K) 

101 for k in range(0, K): 

102 Mat_sweep = Mat_sweep + np.linalg.matrix_power(LHSinv.dot(RHS), k).dot(LHSinv) 

103 ## 

104 # ---> The update formula for this case need verification!! 

105 update = dt * np.kron(L.sweep.coll.weights, Uadv + Cs) 

106 

107 y1 = np.array([1, 0], dtype='complex') 

108 y2 = np.array([0, 1], dtype='complex') 

109 e1 = np.kron(np.ones(nnodes), y1) 

110 stab_fh_1 = y1 + update.dot(Mat_sweep.dot(e1)) 

111 e2 = np.kron(np.ones(nnodes), y2) 

112 stab_fh_2 = y2 + update.dot(Mat_sweep.dot(e2)) 

113 stab_sdc = np.column_stack((stab_fh_1, stab_fh_2)) 

114 

115 # Stability function of backward Euler is 1/(1-z); system is y' = (Cs+Uadv)*y 

116 # stab_ie = np.linalg.inv( np.eye(2) - step.status.dt*(Cs+Uadv) ) 

117 

118 # For testing, insert exact stability function exp(-dt*i*k*(Cs+Uadv) 

119 # stab_fh = la.expm(Cs+Uadv) 

120 

121 dirkts = dirk(Cs + Uadv, dirk_order) 

122 stab_fh1 = dirkts.timestep(y1, 1.0) 

123 stab_fh2 = dirkts.timestep(y2, 1.0) 

124 stab_dirk = np.column_stack((stab_fh1, stab_fh2)) 

125 

126 rkimex = rk_imex(M_fast=Cs, M_slow=Uadv, order=K) 

127 stab_fh1 = rkimex.timestep(y1, 1.0) 

128 stab_fh2 = rkimex.timestep(y2, 1.0) 

129 stab_rk_imex = np.column_stack((stab_fh1, stab_fh2)) 

130 

131 sol_sdc = findomega(stab_sdc) 

132 sol_dirk = findomega(stab_dirk) 

133 sol_rk_imex = findomega(stab_rk_imex) 

134 

135 # Now solve for discrete phase 

136 phase[0, i] = sol_sdc.real / k_vec[i] 

137 amp_factor[0, i] = np.exp(sol_sdc.imag) 

138 phase[1, i] = sol_dirk.real / k_vec[i] 

139 amp_factor[1, i] = np.exp(sol_dirk.imag) 

140 phase[2, i] = sol_rk_imex.real / k_vec[i] 

141 amp_factor[2, i] = np.exp(sol_rk_imex.imag) 

142 

143 rcParams['figure.figsize'] = 1.5, 1.5 

144 fs = 8

147 plt.plot(k_vec, phase[1, :], '-', color='g', linewidth=1.5, label='DIRK(' + str(dirkts.order) + ')')
