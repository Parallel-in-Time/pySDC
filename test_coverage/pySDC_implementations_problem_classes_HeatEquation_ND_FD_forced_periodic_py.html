75 def __get_A(N, nu, dx, ndim, order): 

76 """ 

77 Helper function to assemble FD matrix A in sparse format 

78 

79 Args: 

80 N (list): number of dofs 

81 nu (float): diffusion coefficient 

82 dx (float): distance between two spatial nodes 

83 ndim (int): number of dimensions 

84 

85 Returns: 

86 scipy.sparse.csc_matrix: matrix A in CSC format 

87 """

89 if order == 2: 

90 stencil = [1, -2, 1] 

91 zero_pos = 2 

92 elif order == 4: 

93 stencil = [-1 / 12, 4 / 3, -5 / 2, 4 / 3, -1 / 12] 

94 zero_pos = 3 

95 elif order == 6: 

96 stencil = [1 / 90, -3 / 20, 3 / 2, -49 / 18, 3 / 2, -3 / 20, 1 / 90] 

97 zero_pos = 4 

98 elif order == 8: 

99 stencil = [-1 / 560, 8 / 315, -1 / 5, 8 / 5, -205 / 72, 8 / 5, -1 / 5, 8 / 315, -1 / 560] 

100 zero_pos = 5 

101 else: 

102 raise ProblemError(f'wrong order given, has to be 2, 4, 6, or 8, got {order}') 

103 dstencil = np.concatenate((stencil, np.delete(stencil, zero_pos - 1))) 

104 offsets = np.concatenate(([N[0] - i - 1 for i in reversed(range(zero_pos - 1))], 

105 [i - zero_pos + 1 for i in range(zero_pos - 1, len(stencil))])) 

106 doffsets = np.concatenate((offsets, np.delete(offsets, zero_pos - 1) - N[0])) 

107 

108 A = sp.diags(dstencil, doffsets, shape=(N[0], N[0]), format='csc') 

109 # stencil = [1, -2, 1] 

110 # A = sp.diags(stencil, [-1, 0, 1], shape=(N[0], N[0]), format='csc') 

111 if ndim == 2: 

112 A = sp.kron(A, sp.eye(N[0])) + sp.kron(sp.eye(N[1]), A) 

113 elif ndim == 3: 

114 A = sp.kron(A, sp.eye(N[1] * N[0])) + sp.kron(sp.eye(N[2] * N[1]), A) + \ 

115 sp.kron(sp.kron(sp.eye(N[2]), A), sp.eye(N[0])) 

116 A *= nu / (dx ** 2) 

117 

118 return A

150 """ 

151 Simple linear solver for (I-factor*A)u = rhs 

152 

153 Args: 

154 rhs (dtype_f): right-hand side for the linear system 

155 factor (float): abbrev. for the local stepsize (or any other factor required) 

156 u0 (dtype_u): initial guess for the iterative solver 

157 t (float): current time (e.g. for time-dependent BCs) 

158 

159 Returns: 

160 dtype_u: solution as mesh 

161 """ 

162 

163 me = self.dtype_u(self.init) 

164 

165 if self.params.direct_solver: 

166 me[:] = spsolve(self.Id - factor * self.A, rhs.flatten()).reshape(self.params.nvars) 

167 else: 

168 me[:] = gmres(self.Id - factor * self.A, rhs.flatten(), x0=u0.flatten(), 

169 tol=self.params.lintol, maxiter=self.params.liniter)[0].reshape(self.params.nvars) 

170 return me
